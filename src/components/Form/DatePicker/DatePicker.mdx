import { useState } from 'react';
import { Playground, Props } from 'docz';
import {
    DatePicker,
    Container,
    Segment,
    Row,
    Col,
    Switch,
} from '../../';
import {
    date,
    hasDatePickerSupport,
} from '../../../utils';


# Form | DatePicker
### A simple masked fallback is displayed when browser doesn't have the native picker.


## Examples

<Playground>
    {() => {
        const hasNativeSupport = hasDatePickerSupport();
        const [ examples, setExamples ] = useState({
            first : '',
            second: new Date(),
            third : date().format(), // from 'utils'
        });

        function handleChange(evt, _value) {
            const { target } = evt;
            const { name, value } = target;

            console.log('DatePicker changed', value, _value);

            setExamples({
                ...examples,
                [name]: value,
            });
        }

        return (
            <Container fluid>
                <Row horizontal="center" vertical="start">
                    <Col xs={12}>
                        <Segment padding="10px 0">
                            <DatePicker
                                id="ExampleDatePicker"
                                name="first"
                                onChange={handleChange}
                                value={examples.first}
                            />
                        </Segment>
                    </Col>
                    <Col xs={6}>
                        <Segment padding="10px 0">
                            <DatePicker
                                id="ExampleDatePickerSecond"
                                label="Start"
                                name="second"
                                onChange={handleChange}
                                value={examples.second}
                                max={examples.third}
                            />
                        </Segment>
                    </Col>
                    <Col xs={6}>
                        <Segment padding="10px 0">
                            <DatePicker
                                id="ExampleDatePickerThird"
                                label="End"
                                name="third"
                                onChange={handleChange}
                                value={examples.third}
                                min={examples.second}
                            />
                        </Segment>
                    </Col>
                </Row>
                <Segment padding="30px 0 10px">
                    <Row horizontal="center" vertical="start">
                        <Col xs={6}>
                            <Switch
                                id="ExampleDatePickerSupport"
                                disabled
                                checked={hasNativeSupport}>
                                Browser has native picker support?
                            </Switch>
                        </Col>
                        <Col xs={6}>
                            <Switch
                                id="ExampleDatePickerDisableNative"
                                disabled
                                checked={hasNativeSupport}>
                                And the native picker is active?
                            </Switch>
                        </Col>
                    </Row>
                </Segment>
            </Container>
        );
    }}
</Playground>
