import { memo, useState } from 'react';
import { Playground, Props } from 'docz';
import dayjs from 'dayjs';
import {
    DatePicker,
    Styled,
    Row,
    Col,
} from '../../';

# Form | DatePicker
### Uses browser's datepicker and do some validations. InputDateFallback when Safari Desktop.

This component works with [Day.js](https://day.js.org/), which is a project dependency.

## Examples
<Playground>
    {() => {
        const [example, setExample] = useState();

        return (
            <Row>
                <Col as={Styled} xs={12} sm={6} padding="10px">
                    <div>
                        Regular DatePicker
                    </div>
                    <br />
                    <DatePicker
                        id="example"
                        value={example}
                        onChange={({ target }) => setExample(target.value)}
                        label="Pick a date"
                    />
                    <br />
                    <div>
                        <strong>input value: </strong>
                        {JSON.stringify(example)}
                    </div>
                </Col>
            </Row>
        );
    }}
</Playground>

## Minimum and Maximum
<Playground>

        const [min, setMin] = useState(dayjs().subtract(2, 'days').format('YYYY-MM-DD'));

        return (
            <Row vertical="start">

                        id="exampleFrom"

                        min={min}
