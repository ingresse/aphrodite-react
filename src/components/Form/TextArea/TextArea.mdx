## Delayed

<Playground>
    {() => {
        const [ value, setValue ] = useState('');

        useEffect(() => {
            setTimeout(() => {
                setValue('aphrodite');
            }, 2500);
        }, []);

        function handleChange(evt) {
            setValue(evt.target.value);
        }

        return (
            <Segment padding="10px 0">
                <TextArea
                    label="Big content goes here"
                    placeholder="Placeholder"
                    value={value}
                    onChange={handleChange}
                    rows="8"
                />
            </Segment>
        );
    }}
