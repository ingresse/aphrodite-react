    InputNumber,

# Input Number
### Simple customization of awesome [react-input-number](https://www.npmjs.com/package/react-input-number) component.


## Example

<Playground>
    {() => {
        const [ value, setValue ] = useState('1a2b3c4d5e');

        function handleChange(numberValue) {
            setValue(numberValue);
        }

        return (
            <Container xs>
                <Segment padding="10px 0">
                    <InputNumber
                        id="ExampleInputNumber"
                        label="Type something"
                        onChange={handleChange}
                        value={value}
                    />
                </Segment>
            </Container>
        );
    }}
</Playground>
