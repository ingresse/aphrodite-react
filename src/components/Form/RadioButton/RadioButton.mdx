## Example - Controlled Inputs

<Playground>
    {() => {
        const eventTypes = ['In-person', 'Live', 'Hybrid', 'Invalid'];
        const colSizeSM = parseInt((12 / eventTypes.length), 10);
        const [selectedType, setSelectedType] = useState(eventTypes[0]);

        return (
            <Styled as="section" padding="20px 0">
                <Row>
                    {eventTypes.map((eventType) => {
                        const inputRadioId = `event-type-${eventType}`;

                        return (
                            <Col
                                id={colSizeSM}
                                key={inputRadioId}
                                xs={12}
                                sm={colSizeSM}
                            >
                                <RadioButton
                                    name="event-type"
                                    id={inputRadioId}
                                    value={eventType}
                                    checked={eventType === selectedType}
                                    onChange={({ target }) => setSelectedType(target.value)}
                                    disabled={eventType === eventTypes[eventTypes.length - 1]}
                                    description="controlled input"
                                >
                                    {eventType}
                                </RadioButton>
                            </Col>
                        );
                    })}
                </Row>
            </Styled>
        );
    }}
</Playground>
