#### Delayed
<Playground>
    {() => {
        const [ value, setValue ] = useState('');

        useEffect(() => {
            setTimeout(() => {
                setValue('aphrodite');
            }, 2500);
        }, []);

        function handleChange(evt) {
            setValue(evt.target.value);
        }

        return (
            <Select
                label="Select Label"
                value={value}
                onChange={handleChange}>
                <option value=""></option>
                <option value="ingresse">Ingresse</option>
                <option value="aphrodite">Aphrodite</option>
                <option value="react">React</option>
            </Select>
        );
    }}
</Playground>

## Switch
<Props of={Switch} />

### Example
<Playground>
    {() => {
        const [ col, setCol ]                 = useState(4);
        const [ text, setText ]               = useState('Switch Text');
        const [ disabled, setDisabled ]       = useState(false);
        const [ textAlign, setTextAlign ]     = useState(false);
        const [ textSmall, setTextSmall ]     = useState(false);
        const [ toggleAlign, setToggleAlign ] = useState(false);

        const BreakPointToggle = (props) => (
            <Placeholder
                {...props}
                block
                height={20}
                styles={{ cursor: 'pointer' }}
                title={`Toggle width to ${props.size} columns`}
                onClick={() => setCol(props.size)}
            />
        );

        return (
            <Segment padding="0" >
                {(window.innerWidth < 768) ? (null) : (
                    <Segment padding="0 0 10px">
                        <Row horizontal="center">
                            <Col xs={12} sm={2}>
                                <BreakPointToggle size={12} />
                            </Col>
                            <Col xs={12} sm={2}>
                                <BreakPointToggle size={8} />
                            </Col>
                            <Col xs={12} sm={4}>
                                <BreakPointToggle size={4} />
                            </Col>
                            <Col xs={12} sm={2}>
                                <BreakPointToggle size={8} />
                            </Col>
                            <Col xs={12} sm={2}>
                                <BreakPointToggle size={12} />
                            </Col>
                        </Row>
                    </Segment>
                )}

                <Segment padding="20px 0">
                    <Row horizontal="center">
                        <Col xs={12} sm={col}>
                            <Card>
                                <Switch
                                    disabled={disabled}
                                    textSmall={textSmall}
                                    textAlign={textAlign ? 'right' : 'left'}
                                    toggleAlign={toggleAlign ? 'right' : 'left'}>
                                    {text}
                                </Switch>
                            </Card>
                        </Col>
                    </Row>

                    <Segment padding="40px 0">
                        <Row horizontal="center">
                            <Col xs={12} sm={12} md={9}>
                                <Segment padding="10px 0">
                                    <Input
                                        type="text"
                                        label="Switch Label Text"
                                        placeholder="Type something..."
                                        value={text}
                                        onChange={(e) => setText(e.target.value)}
                                    />
                                </Segment>
                            </Col>
                        </Row>

                        <Row horizontal="center">
                            <Col xs={6} sm={6} md={3}>
                                <Segment padding="10px 0">
                                    <Checkbox
                                        id="switchToggleAlign"
                                        value={toggleAlign ? 'right' : 'left'}
                                        defaultChecked={toggleAlign}
                                        onChange={(e) => setToggleAlign(e.target.checked ? true : false)}>
                                        Align toggle at right
                                    </Checkbox>
                                </Segment>
                            </Col>

                            <Col xs={6} sm={6} md={3}>
                                <Segment padding="10px 0">
                                    <Checkbox
                                        id="switchTextAlign"
                                        value={textAlign ? 'right' : 'left'}
                                        defaultChecked={textAlign}
                                        onChange={(e) => setTextAlign(e.target.checked ? true : false)}>
                                        Align text at right
                                    </Checkbox>
                                </Segment>
                            </Col>

                            <Col xs={6} sm={6} md={3}>
                                <Segment padding="10px 0">
                                    <Checkbox
                                        id="switchTextSmall"
                                        value={textSmall ? true : false}
                                        defaultChecked={textSmall}
                                        onChange={(e) => setTextSmall(e.target.checked ? true : false)}>
                                        Text small
                                    </Checkbox>
                                </Segment>
                            </Col>

                            <Col xs={6} sm={6} md={3}>
                                <Segment padding="10px 0">
                                    <Checkbox
                                        id="switchDisabled"
                                        value={disabled ? true : false}
                                        defaultChecked={disabled}
                                        onChange={(e) => setDisabled(e.target.checked ? true : false)}>
                                        Disabled
                                    </Checkbox>
                                </Segment>
                            </Col>
                        </Row>
                    </Segment>
                </Segment>
            </Segment>
        );
    }}
</Playground>
