```jsx
import { Segment } from 'aphrodite-react';

...
```

## Examples
<Playground style={{ padding: '20px' }}>
    {() => {
        const customStyles = {
            color     : colors.get('white'),
            background: colors.get('bamboo'),

            borderRadius: RADIUS.SM,
        };

        const [ segmentStyles, setSegmentStyles ] = useState(customStyles);

        const toggleStyles = () => {
            setSegmentStyles(segmentStyles ? null : customStyles);
        };

        return (
            <Container>
                <Row horizontal="center">
                    <Col xs={12} sm={6}>
                        <Segment>
                            Default Segment
                        </Segment>

                        <Segment color={colors.get('supernova')}>
                            Segment with custom color prop (NEW)
                        </Segment>

                        <Segment styles={segmentStyles}>
                            <div>
                                This is a `Segment`.
                            </div>
                            A simple block to segmentate all the things.
                        </Segment>

                    </Col>
                </Row>

                <Segment padding="40px 0 10px">
                    <Row horizontal="center">
                        <Col xs={12} sm={6}>
                            <Card>
                                <Switch
                                    value={segmentStyles ? true : false}
                                    defaultChecked={segmentStyles ? true : false}
                                    onChange={toggleStyles}>
                                    Display with custom styles.
                                </Switch>
                            </Card>
                        </Col>
                    </Row>
                </Segment>
            </Container>
        );
    }}
</Playground>
