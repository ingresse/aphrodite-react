import { colors } from '../../utils';

<Playground>
    {() => {
        /**
         * Local values
         */
        const [ size, setSize ]   = useState(30);
        const [ color, setColor ] = useState('none');
        const [ term, setTerm ]   = useState('');

        return (
            <>
                <form onSubmit={(evt) => evt.preventDefault()}>
                    <Segment padding="0">
                        <Row vertical="middle" horizontal="center">
                            <Col xs={12} sm={8} first="sm" last="xs">
                                <Segment padding="0 0 20px">
                                    <Input
                                        id="aph_icons_search"
                                        type="search"
                                        autoComplete="off"
                                        label="Search Icons"
                                        value={term}
                                        onChange={(evt, value) => setTerm(value)}
                                    />
                                </Segment>
                            </Col>
                            <Col xs={6} sm={2}>
                                <Segment padding="0 0 20px">
                                    <Select
                                        id="aph_icons_size"
                                        label="Size"
                                        value={size}
                                        onChange={(evt, value) => setSize(value)}>
                                        <option value={20}>
                                            20
                                        </option>
                                        <option value={30}>
                                            30
                                        </option>
                                        <option value={40}>
                                            40
                                        </option>
                                    </Select>
                                </Segment>
                            </Col>
                            <Col xs={6} sm={2}>
                                <Segment padding="0 0 20px">
                                    <Select
                                        id="aph_icons_color"
                                        label="Color"
                                        value={color}
                                        onChange={(evt, value) => setColor(value)}>
                                        <option value={'none'}>
                                            random
                                        </option>
                                        {Object.keys(colors.shades).map((colorKey) => (
                                            <option key={`aph_icons_color_${colorKey}`} value={colorKey}>
                                                {colorKey}
                                            </option>
                                        ))}
                                    </Select>
                                </Segment>
                            </Col>
                        </Row>
                    </Segment>

                    <Row vertical="middle" horizontal="center">
                        {ICONS.map((ICON_SLUG, INDEX) => {
                            if (term && !ICON_SLUG.toLowerCase().includes(term.toLowerCase())) {
                                return (null);
                            }

                            const IconRendered = (props) => (
                                <Icon
                                    slug={ICON_SLUG}
                                    size={parseInt(size, 10)}
                                    color={color !== 'none' ? color : ((INDEX % 2) ? 'primary' : 'secondary')}
                                    {...props}
                                />
                            );

                            const iconMarkup =
                                `<Icon slug="${ICON_SLUG}" size={${size}} color="${color !== 'none' ? color : ((INDEX % 2) ? 'primary' : 'secondary')}" />`
                            ;

                            return (
                                <Col xs={12} sm={4} md={3} key={`aph_icon_${INDEX}`}>
                                    <HoverActions
                                        styled
                                        styles={{ display: 'block', width: 'auto' }}
                                        actions={(
                                            <>
                                                <Clipboard
                                                    styled
                                                    data-clipboard-text={ICON_SLUG}>
                                                    Copy Slug
                                                </Clipboard>
                                                <Clipboard
                                                    styled
                                                    data-clipboard-text={iconMarkup}>
                                                    Copy Markup
                                                </Clipboard>
                                            </>
                                        )}>
                                        <Card
                                            hover
                                            boxShadow
                                            styles={{
                                                margin: '0 0 20px',
                                                backgroundColor: ((color === 'translucid') ? colors.get('oil') : null),
                                            }}>
                                            <Row vertical="middle" horizontal="center">
                                                <Col xs={4} sm={4} lg={3} styles={{ textAlign: 'center' }}>
                                                    <IconRendered />
                                                </Col>
                                                <Col xs={8} sm={8} lg={9}>
                                                    <Text bold small center helper>
                                                        {ICON_SLUG}
                                                    </Text>
                                                </Col>
                                            </Row>
                                        </Card>
                                    </HoverActions>
                                </Col>
                            );
                        })}
                    </Row>
                </form>
            </>
        );
    }}
</Playground>
