# Icon
#### Simple SVG icons

```jsx
import { Icon } from 'aphrodite-react';
...
<Icon slug="arrow-up-circle" size={40} color="ruby" />
```

<Props of={Icon} />

## Example
<Playground>
    {() => {
        return ICONS.map((ICON_SLUG, INDEX) => {
            return (
                <div key={INDEX} style={{ margin: '5px 0' }}>
                    <Icon slug={ICON_SLUG} size={20 + ((INDEX + 1) * 5)} color={(INDEX % 2) ? 'ruby' : ''} />
                    <span style={{ paddingLeft: '10px' }}>{ICON_SLUG}</span>
                </div>
            );
        });
    }}
</Playground>
