# Dropdown
#### A white block, rounded, with simple shadow effect, downing from some other component.

```jsx
import { Dropdown } from 'aphrodite-react';

...

<Dropdown>
    Hello
</Dropdown>
```

## Advanced Examples
<Playground style={{ padding: '20px' }}>
    {() => {
        const [ dropdownVisible, setDropdownVisible ] = useState(false);
        const [ dropdownValue, setDropdownValue ]     = useState('Dropdown Item B');

        return (
            <Segment padding="40px 0 120px">
                <Row>
                    <Col xs={12} sm={6}>
                        <Segment padding="20px 0">
                            <H3 styles={{ marginBottom: '25px' }}>
                                Custom Toggle
                            </H3>
                            <Dropdown
                                width="300px"
                                toggleBlock
                                toggle={(
                                    <Button block>
                                        Custom Toggle
                                    </Button>
                                )}>
                                This is the dropdown content, triggered by a custom toggle
                            </Dropdown>
                        </Segment>
                    </Col>
                    <Col xs={12} sm={6}>
                        <Segment padding="20px 0">
                            <H3 styles={{ marginBottom: '25px' }}>
                                Focus
                            </H3>
                            <Dropdown
                                thin
                                opened={dropdownVisible}
                                toggleBlock
                                toggle={(
                                    <Input
                                        id="MyInputAsDropdownToggle"
                                        label="Selecting a Value from Dropdown"
                                        tabIndex={20}
                                        value={dropdownValue}
                                        onChange={(evt) => setDropdownValue(evt.target.value || '')}
                                        onBlur={() => { setDropdownVisible(false) }}
                                        onFocus={() => { setDropdownVisible(true) }}
                                    />
                                )}>
                                <List>
                                    <ListItem>
                                        Type or Select one option
                                    </ListItem>
                                    <ListItem
                                        className={dropdownValue === 'Dropdown Item A' ? 'active' : ''}
                                        onClick={() => setDropdownValue('Dropdown Item A')}>
                                        Dropdown Item A
                                    </ListItem>
                                    <ListItem
                                        className={dropdownValue === 'Dropdown Item B' ? 'active' : ''}
                                        onClick={() => setDropdownValue('Dropdown Item B')}>
                                        Dropdown Item B
                                    </ListItem>
                                </List>
                            </Dropdown>
                        </Segment>
                    </Col>
                </Row>
            </Segment>
        );
    }}
</Playground>
