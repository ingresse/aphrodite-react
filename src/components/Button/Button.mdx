## Colors Shades

<Playground>
    {() => {
        const colorsShades   = Object.keys(colors.shades);
        const colorsExcluded = [
            'primary',
            'secondary',
            'info',
            'success',
            'warning',
            'error',
            'translucid',
        ];

        return (
            <Row horizontal="center">
                {colorsShades.map((colorKey) => {
                    if (colorsExcluded.includes(colorKey)) {
                        return (null)
                    }

                    return (
                        <Col xs={6} sm={4} md={4} lg={3} key={`color-${colorKey}`}>
                            <Button block margin="10px 0" color={colorKey}>
                                {colorKey}
                            </Button>
                            <Button block small margin="10px 0" color={colorKey}>
                                {colorKey}
                            </Button>
                        </Col>
                    );
                })}
            </Row>
        );
    }}
</Playground>

## Loading State

<Playground>
    {() => {
        const [ btnText, setBtnText ] = useState('Load');
        const [ loading, setLoading ] = useState(false);

        function toggleLoading() {
            setLoading(!loading);

            setTimeout(() => {
                setBtnText('Done');
                setLoading(false);

                setTimeout(() => {
                    setBtnText('Again');
                }, 1000);
            }, 5000);
        }

        return (
            <Row horizontal="center">
                <Col xs={12} sm={6}>
                    <Styled textAlign="center">
                        <Button onClick={toggleLoading} loading={loading}>
                            {btnText}
                        </Button>
                    </Styled>
                </Col>
                <Col xs={12} sm={6}>
                    <Styled textAlign="center">
                        <Button small onClick={toggleLoading} loading={loading}>
                            {btnText}
                        </Button>
                    </Styled>
                </Col>
            </Row>
        );
    }}
</Playground>
