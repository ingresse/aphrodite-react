## Colors Shades

<Playground>
    {() => {
        const colorsShades   = Object.keys(colors.shades);
        const colorsExcluded = [
            'primary',
            'secondary',
            'info',
            'success',
            'warning',
            'error',
            'translucid',
        ];

        return (
            <Row horizontal="center">
                {colorsShades.map((colorKey) => {
                    if (colorsExcluded.includes(colorKey)) {
                        return (null)
                    }

                    return (
                        <Col xs={6} sm={4} md={3} lg={2} key={`color-${colorKey}`}>
                            <Button block margin="10px 0" color={colorKey}>
                                {colorKey}
                            </Button>
                            <Button block small margin="10px 0" color={colorKey}>
                                {colorKey}
                            </Button>
                        </Col>
                    );
                })}
            </Row>
        );
    }}
</Playground>
