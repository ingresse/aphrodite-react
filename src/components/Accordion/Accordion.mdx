# Accordion
#### Toggle sub-items visibility.
```jsx
import { Accordion, ListGroup } from 'aphrodite-react';

<ListGroup>
    <Accordion header="ANYTHING">
        ...
    </Accordion>
</ListGroup>
```

## Nested
<Playground>
    {() => {
        const [ opened, setOpened ] = useState(false);

        return (
            <Segment padding="0">
                <Segment
                    padding="0 0 20px"
                    styles={{ textAlign: 'center' }}>
                    <Button
                        type="button"
                        onClick={() => setOpened(!opened)}>
                        Toggle from external value
                    </Button>
                    <Segment padding="15px 0 0">
                        Should be opened: {opened ? 'YES' : 'NO'}
                    </Segment>
                </Segment>

                <ListGroup>
                    <Accordion
                        opened={opened}
                        openedCallback={setOpened}
                        headerDark
                        headerStyles={{ color: 'red' }}
                        header={
                            <div>
                                <strong>GRAND</strong>FATHER
                            </div>
                        }>
                        <div>
                            <Accordion header="FATHER">
                                <div style={{ padding: '10px 15px' }}>
                                    Accordion Children
                                </div>
                            </Accordion>
                        </div>
                    </Accordion>
                </ListGroup>
            </Segment>
        );
    }}
</Playground>
