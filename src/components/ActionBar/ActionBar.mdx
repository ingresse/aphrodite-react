# Action Bar
#### Toggle visibility of action elements
```jsx
import { ActionBar, Button } from 'aphrodite-react';

<ActionBar
    visible={true}
    styles={{
        textAlign: 'center',
    }}>
    <Button>My Action</Button>
</ActionBar>
```

## Example
<Playground>
    {() => {
        const [ visible, setVisible ] = useState(true);

        function toggle() {
            setVisible(!visible);
        }

        return (
            <Segment
                style={{
                    position    : 'relative',
                    minHeight   : '240px',
                    border      : '1px solid rgba(0, 0, 0, 0.1)',
                    borderRadius: '4px',
                    overflow    : 'hidden',
                    padding     : '15px 0 0',
                    textAlign   : 'center',
                }}>
                <Segment>
                    The Page Container
                </Segment>

                <Button
                    link
                    onClick={toggle}>
                    Toggle Action Bar
                </Button>

                <ActionBar
                    visible={visible}
                    styles={{
                        position : 'absolute', // Just to fit in this example. Default: 'fixed',
                    }}>
                    <Button
                        color="primary"
                        onClick={toggle}>
                        Close Action Bar
                    </Button>
                </ActionBar>
            </Segment>
        );
    }}
</Playground>
