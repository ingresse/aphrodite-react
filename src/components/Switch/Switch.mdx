## onChange
<Playground>
    {() => {
        let check = false;
        return (
            <div>
                <ToastsContainer styles={{top: '45px', right: '45px'}} />
                <Switch checked={check} onChange={
                    () => {
                        check = !check;
                        if (check) {
                            toast.success("Switch Activated");
                            return;
                        }
                        toast.error("Switch disabled");
                    }
                } />
            </div>
        )
    }}
</Playground>
