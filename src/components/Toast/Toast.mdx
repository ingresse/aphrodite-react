# Toast
#### Just an implementation of awesome library [**React Toastify**](https://github.com/fkhadra/react-toastify), with Aphrodite's definitions.
#### So, it's just consult the [documentation](https://github.com/fkhadra/react-toastify) for more options.

## Usage
<Playground>
    {() => {
        // /* Import */
        // import { Button, ToastsContainer, toast } from 'aphrodite-react';
        //

        function toastDefault() {
            toast("Hello from Aphrodite React");
        }

        function toastInverse() {
            toast.inverse("Hello from Aphrodite React");
        }

        function toastSuccess() {
            toast.success("We are using React Toastify, an awesome library, customized with our styles definitions");
        }

        function toastInfo() {
            toast.info("You can use React Toastify directly, if you want");
        }

        function toastWarning() {
            toast.warning("But, isn't have our style, if you use directly");
        }

        function toastError() {
            toast.error("And maybe doesn't make sense use our library, if you want to use React Toastify directly - haha");
        }

        return (
            <div>
                <ToastsContainer hideProgressBar={true} autoClose={false} styles={{top: '45px', right: '45px'}} />

                <Button color="grey" onClick={toastDefault}>Default</Button>
                <Button color="black" onClick={toastInverse}>Inverse</Button>

                <br />
                <br />

                <Button color="success" onClick={toastSuccess}>Success</Button>
                <Button color="primary" onClick={toastInfo}>Info</Button>
                <Button color="warning" onClick={toastWarning}>Warning</Button>
                <Button color="error" onClick={toastError}>Error</Button>
            </div>
        );
    }}
</Playground>
