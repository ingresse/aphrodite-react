import { useState, useEffect } from 'react';
import { Playground, Props } from 'docz';
import { Pagination, Segment, Text } from '../';


# Pagination
### A styled customization of awesome library [RC Pagination](https://www.npmjs.com/package/rc-pagination).
### Check it out there to more usage details.


## Example

<Playground>
    {() => {
        const [ timer, setTimer ]           = useState(null);
        const [ pagination, setPagination ] = useState({
            loading : false,
            total   : 50,
            current : 5,
            pageSize: 5,
        });

        /**
         * Handle with Page changes,
         * simulating a HTTP request delay
         *
         * @param {number} selectedPage
         */
        function handlePageChange (selectedPage = 1) {
            console.log('Page Selected', selectedPage);

            clearTimeout(timer);

            setPagination({
                ...pagination,
                loading: true,
            });

            setTimer(setTimeout(() => {
                setPagination({
                    ...pagination,
                    loading: false,
                    current: selectedPage
                })
            }, 500));
        }

        return (
            <Segment padding="40px 0">
                <Text center bold>
                    Pagination state object
                </Text>
                <Text center margin="0">
                    {JSON.stringify(pagination)}
                </Text>

                <Pagination
                    {...pagination}
                    onChange={handlePageChange}
                />
            </Segment>
        );
    }}
</Playground>
