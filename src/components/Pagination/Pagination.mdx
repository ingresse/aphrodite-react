# Pagination
#### An simple implementation of awesome library [RC Pagination](http://react-component.github.io/pagination/), check it out there to more usage details.

## Properties
<Props of={Pagination} />

## Example
<Playground>
    {() => {
        const [ pagination, setPagination ] = useState({
            loading : false,
            total   : 50,
            current : 5,
            pageSize: 5,
        });

        function handlePageChange (selectedPage = 1) {
            console.log('Page Selected', selectedPage);

            setPagination({
                ...pagination,
                current: selectedPage
            });
        }

        return (
            <Pagination
                {...pagination}
                onChange={handlePageChange}
            />
        );
    }}
</Playground>
