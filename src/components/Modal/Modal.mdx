<Playground style={{ margin: '-20px', padding: '0 !important' }}>
    {() => {
        // import { Modal, Segment, Button, Placeholder } from 'aphrodite-react';

        const [ opened, setOpened ] = useState(false);

        /** All the `styles` in Modal was only customized to this example */
        return (
            <Segment
                padding="0"
                styles={{
                    position: 'relative',
                    overflow: 'hidden',
                    height  : '530px'
                }}>

                <Segment padding="80px 0" styles={{ textAlign: 'center' }}>
                    <Button
                        color="primary"
                        onClick={() => setOpened(true)}>
                        Open Modal
                    </Button>
                </Segment>

                <Modal
                    title="Oi, eu sou o Goku"
                    opened={opened}
                    openedCallback={setOpened}
                    styles={{
                        position: 'absolute'
                    }}
                    footerProps={{
                        styles: {
                            position : 'absolute',
                            textAlign: 'center',
                        },
                    }}
                    footer={
                        <Button onClick={() => setOpened(false)}>
                            Close Modal
                        </Button>
                    }>
                    <Container sm>
                        <Placeholder>
                            <iframe
                                width="100%"
                                height="325"
                                src="https://www.youtube-nocookie.com/embed/oyTZ267Xg70?autoplay=true&volume=0.3&start=4"
                                frameBorder="0"
                                allow="accelerometer; autoplay; encrypted-media; gyroscope; picture-in-picture"
                                allowFullScreen
                            />
                        </Placeholder>
                    </Container>
                </Modal>
            </Segment>
        );
    }}
