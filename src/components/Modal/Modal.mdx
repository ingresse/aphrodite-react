import { useState } from 'react';
import { Modal, Button, Row, Segment } from '../';

import { Playground, Props } from 'docz';

## Properties
<Props of={Modal} />

## Example
<Playground style={{ margin: '-20px', padding: '0 !important' }}>
    {() => {
        const [ opened, setOpened ] = useState(false);

        return (
            <Segment
                padding="0"
                styles={{
                    position: 'relative',
                    overflow: 'hidden',
                    height  : '520px'
                }}>

                <Segment padding="80px 0" styles={{ textAlign: 'center' }}>
                    <Button
                        color="primary"
                        onClick={() => setOpened(true)}>
                        Open Modal
                    </Button>
                </Segment>

                <Modal
                    title="Modal Title"
                    opened={opened}
                    openedCallback={setOpened}
                    styles={{ position: 'absolute' }}
                    footerProps={{
                        styles: {
                            position: 'absolute',
                        },
                    }}
                    footer={
                        <Row horizontal="center">
                            <Button onClick={() => setOpened(false)}>
                                Close Modal
                            </Button>
                        </Row>
                    }>
                    Modal Content
                </Modal>
            </Segment>
        );
    }}
</Playground>
