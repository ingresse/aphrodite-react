# Modal
#### Show and hide items in modal

```jsx
import { Modal } from 'aphrodite-react';
...
    <Modal
        title="Modal title"
        opened={true}
        footer={<Component />}>
        <Component />
    </Modal>
```

## Example
<Playground style={{ margin: '-20px', padding: '0 !important' }}>
    {() => {
        const [ opened, setOpened ] = useState(false);

        function toggleModal() {
            setOpened(!opened);
        }

        return (
            <Segment
                padding="0"
                styles={{
                    position: 'relative',
                    overflow: 'hidden',
                    height  : '640px'
                }}>

                <Segment styles={{ textAlign: 'center' }}>
                    <Button color="primary" onClick={toggleModal}>
                        Open Modal
                    </Button>
                </Segment>

                <Modal
                    title="Modal Title"
                    opened={opened}
                    styles={{ position: 'absolute' }}
                    footerProps={{
                        styles: {
                            position: 'absolute',
                        },
                    }}
                    footer={
                        <Row horizontal="center">
                            <Button onClick={toggleModal}>
                                Close Modal
                            </Button>
                        </Row>
                    }>
                    Modal Content
                </Modal>
            </Segment>
        );
    }}
</Playground>
