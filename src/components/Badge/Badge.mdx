## Colors
<Playground>
    {() => {
        const comStyle = {
            boxSizing: 'border-box',

            '&:after': {
                display: 'table',
                clear  : 'both',
                content: '" "',
                boxSizing: 'border-box',
            },
        };

        const rowStyle = Object.assign({
            display: 'block',
            margin : '10px -10px',
        }, comStyle);

        const colStyle = Object.assign({
            display: 'block',
            float  : 'left',
            width  : window.innerWidth < SCREEN_SIZES.SM ? '100%' : '25%',
            padding: '10px',
        }, comStyle);

        const Row   = styled('div')(rowStyle);
        const Col   = styled('div')(colStyle);
        const width = window.innerWidth < SCREEN_SIZES.SM ? '100%' : '150px';

        return (
            <Row>
                {Object.keys(COLORS).map((COLOR_KEY) => {
                    if (!['GET', 'FILL', 'TONES'].includes(COLOR_KEY)) {
                        return (
                            <Col>
                                <Row>
                                    <Badge width={width} color={COLOR_KEY} prefix="KEY">
                                        {COLOR_KEY}
                                    </Badge>
                                </Row>
                                <Row>
                                    <Badge width={width} color={COLOR_KEY} prefix="KEY" sm>
                                        {COLOR_KEY}
                                    </Badge>
                                </Row>
                            </Col>
                        );
                    }

                    return null;
                })}
            </Row>
        );
    }}
</Playground>
