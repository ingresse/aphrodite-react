# Aphrodite
### React UI Library from [ingresse.com](https://www.ingresse.com)

---

## Install
#### You can install with Yarn or NPM in your App Project, but using GitHub repository address as source:
```
yarn add https://github.com/ingresse/aphrodite-react.git
```

---

## Usage Example
#### Include Aphrodite Components inside your React application:

```jsx
/* Core Packages */
import React from 'react';
import propTypes from 'prop-types';

/* UI Library Components */
import { Container, Button } from 'aphrodite-react';

/* Component Itself */
const MyComponent = (props) = {
    const { handleClick } = props;

    return (
        <Container>
            <Button onClick={handleClick}>
                Aphrodite Button
            </Button>
        </Container>
    );
};

/* Component Default Properties */
MyComponent.defaultProps = {
    /**
     * Handle with Button Click
     *
     * @param {object} e - Virtual DOM Synthetic Click Event
     */
    handleClick: (e) => { console.log('handleClick acting', e) },
};

/* Component Properties Types */
MyComponent.propTypes = {
    handleClick: propTypes.func.isRequired,
};

/* Exporting */
export default MyComponent;

```
