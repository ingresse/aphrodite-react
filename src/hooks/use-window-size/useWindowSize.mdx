# Hook | useWindowSize
### Simple hook to return window width and height

```jsx
import {
    useWindowSize,
    H1,
    H3,
    Container,
    Segment,
} from 'aphrodite-react';

function Header() {
    const {
        xs,
        sm,
        md,
        lg,
        xl,
        width,
        height,
    } = useWindowSize();

    const Title = (xs ? H3 : H1);

    return (
        <Segment
            as="header"
            padding={xs ? '20px 0' : '40px 0'}>
            <Container>
                <Title>
                    Sales Report
                </Title>
            </Container>
        </Segment>
    );
}
```
